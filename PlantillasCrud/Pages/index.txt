<script setup lang="ts">
import { manejaError } from '@/utils/funcionesComunes'

definePageMeta({
  middleware: 'permissions',
  action: 'Listar {{ modeloTitleCase }}',
  subject: '{{ model }}',
})

const { can } = useAbility()
const { del } = useClienteRequest()
const { success, preguntaEliminar } = useToast()
const dataTable = ref<any>(null)

const headers = {{ headers }}

const deleteItem = async (id: number) => {
  const confirm = await preguntaEliminar('¿Desea Eliminar {{ model }}?')
  if (!confirm)
    return
  try {
    const respuesta = await del(`{{ url }}/${id}`)

    success(respuesta.message)
    if (dataTable.value)
      dataTable.value.getItems()
  }
  catch (errorCarpturado) {
    manejaError(errorCarpturado)
  }
}
</script>

<template>
  <div class="d-flex flex-wrap justify-end justify-sm-space-between gap-y-4 gap-x-6 mb-4">
    <h1 v-text="'Listado de {{ modeloTitleCase }}'"/>
    <VBtn
      v-if="can('Crear {{ modeloTitleCase }}', '{{ model }}')"
      class="ml-auto"
      color="success"
      to="/{{ directory }}/create"
    >
      <VIcon class="mr-2 ri-add-large-fill"/>
      Nuev@ {{ model }}
    </VBtn>
  </div>
  <DataTableComponent
    ref="dataTable"
    :columnas="headers"
    endpoint="{{ url }}"
    :cantidad-por-pagina="10"
    :cantidad-por-pagina-opciones="[10, 20, 30]"
    :botones="['xlsx', 'pdf', 'csv', 'reiniciar']"
  >
    <template #item.Acciones="{ item }">
      <VBtn
        v-if="can('Ver {{ modeloTitleCase }}', '{{ model }}')"
        icon="ri-eye-line"
        variant="tonal"
        color="info"
        :to="`/{{ directory }}/show/${item.id}` "
        class="mr-1"
      />
      <VBtn
        v-if="can('Editar {{ modeloTitleCase }}', '{{ model }}')"
        icon="ri-edit-box-line"
        variant="tonal"
        color="warning"
        :to="`/{{ directory }}/edit/${item.id}`"
        class="mr-1"
      />
      <VBtn
        v-if="can('Eliminar {{ modeloTitleCase }}', '{{ model }}')"
        icon="ri-delete-bin-line"
        variant="tonal"
        color="error"
        @click="deleteItem(item.id)"
      />
    </template>
  </DataTableComponent>
</template>
