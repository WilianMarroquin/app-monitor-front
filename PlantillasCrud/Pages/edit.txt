<script setup lang="ts">
import Fields from '@/views/pages/{{ directory }}/fields.vue'
import type { {{ model }}Interface } from '@/types/{{ directory }}/types'
import { manejaError } from '@/utils/funcionesComunes'
import type { SendResponseInterface } from '@/types/generales/types'

definePageMeta({
  middleware: 'permissions',
  action: 'Editar {{ modeloTitleCase }}',
  subject: '{{ model }}',
})

const { put, get } = useClienteRequest()
const { success } = useToast()
const { paginaEspera } = useCargandoPagina()

const route = useRoute()
const id = route.params.id

const actualizar{{ model }} = async ({{ model }}: {{ model }}Interface): Promise<void> => {
  paginaEspera.value = true
  try {
    const respuesta = await put<SendResponseInterface<{{ model }}Interface>>(`{{ url }}/${id}/`, {{ model }})

    success(respuesta.message)
    navigateTo('/{{ directory }}')
  }
  catch (errorCarpturado: any) {
    manejaError(errorCarpturado)
  }
  finally {
    paginaEspera.value = false
  }
}

const item{{ model }} = ref<{{ model }}Interface>(
{{ camposFormCreate }}
)

const get{{ model }} = async () => {
  paginaEspera.value = true
  try {
    const respuesta = await get<SendResponseInterface<{{ model }}Interface>>(`{{ url }}/${id}/`)
    item{{ model }}.value = respuesta.data
  }
  catch (errorCarpturado: any) {
    manejaError(errorCarpturado)
  }
  finally {
    paginaEspera.value = false
  }
}

get{{ model }}()

const puedeMostrarDatos = computed(() => {
  return Object.values(item{{ model }} .value).some(valor => valor !== null && valor !== undefined)
})
</script>

<template>

  <div class="d-flex flex-wrap justify-end justify-sm-space-between gap-y-4 gap-x-6 mb-6">
    <h1 v-text="'Editar {{ model }}'"/>
    <VBtn
      class="ml-auto"
      color="secondary"
      to="/{{ directory }}/"
    >
      <VIcon class="mr-2 ri-contract-left-fill"/>
      Regresar
    </VBtn>

  </div>

  <VCard>
    <VCardText>
      <Fields
        v-if="puedeMostrarDatos"
        :item="item{{ model }}"
        @emitir-datos="actualizar{{ model }}"
      />
    </VCardText>
  </VCard>
</template>
